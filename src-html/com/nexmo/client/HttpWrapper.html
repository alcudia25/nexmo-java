022package com.nexmo.client;
023
024import com.nexmo.client.auth.AuthCollection;
025import org.apache.http.client.HttpClient;
026import org.apache.http.client.config.RequestConfig;
027import org.apache.http.config.ConnectionConfig;
028import org.apache.http.config.SocketConfig;
029import org.apache.http.impl.client.DefaultHttpClient;
030import org.apache.http.impl.client.HttpClientBuilder;
031import org.apache.http.impl.conn.PoolingHttpClientConnectionManager;
032import org.apache.http.impl.conn.tsccm.ThreadSafeClientConnManager;
033import org.apache.http.params.BasicHttpParams;
034import org.apache.http.params.HttpConnectionParams;
035import org.apache.http.params.HttpParams;
036import org.apache.http.params.HttpProtocolParams;
037
038import java.nio.charset.Charset;
039
040/**
041 * Internal class that holds available authentication methods and a shared HttpClient.
042 */
043public class HttpWrapper {
044    private AuthCollection authMethods;
045    private HttpClient httpClient = null;
046
047    public HttpWrapper(AuthCollection authMethods) {
048        this.authMethods = authMethods;
049    }
050
051    public HttpClient getHttpClient() {
052        if (this.httpClient == null) {
053            this.httpClient = createHttpClient();
054        }
055        return this.httpClient;
056    }
057
058    public void setHttpClient(HttpClient httpClient) {
059        this.httpClient = httpClient;
060    }
061
062    public AuthCollection getAuthMethods() {
063        return authMethods;
064    }
065
066    public void setAuthMethods(AuthCollection authMethods) {
067        this.authMethods = authMethods;
068    }
069
070    protected HttpClient createHttpClient() {
071        PoolingHttpClientConnectionManager connectionManager = new PoolingHttpClientConnectionManager();
072        connectionManager.setDefaultMaxPerRoute(200);
073        connectionManager.setMaxTotal(200);
074        connectionManager.setDefaultConnectionConfig(
075                ConnectionConfig.custom().setCharset(Charset.forName("UTF-8")).build());
076        connectionManager.setDefaultSocketConfig(SocketConfig.custom().setTcpNoDelay(true).build());
077
078        // Need to work out a good value for the following:
079        // threadSafeClientConnManager.setValidateAfterInactivity();
080
081        RequestConfig requestConfig = RequestConfig.custom().build();
082
083        return HttpClientBuilder.create()
084                .setConnectionManager(connectionManager)
085                .setUserAgent("nexmo-java/2.0.0-prerelease")
086                .setDefaultRequestConfig(requestConfig)
087                .build();
088    }
089}

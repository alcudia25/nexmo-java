022package com.nexmo.client.auth;
023
024import com.auth0.jwt.Algorithm;
025import com.auth0.jwt.JWTSigner;
026import com.nexmo.client.NexmoUnexpectedException;
027import org.apache.http.client.methods.RequestBuilder;
028
029import javax.xml.bind.DatatypeConverter;
030import java.io.IOException;
031import java.io.UnsupportedEncodingException;
032import java.nio.file.Files;
033import java.nio.file.Path;
034import java.security.InvalidKeyException;
035import java.security.KeyFactory;
036import java.security.NoSuchAlgorithmException;
037import java.security.PrivateKey;
038import java.security.spec.InvalidKeySpecException;
039import java.security.spec.PKCS8EncodedKeySpec;
040import java.util.HashMap;
041import java.util.Map;
042import java.util.UUID;
043import java.util.regex.Matcher;
044import java.util.regex.Pattern;
045
046public class JWTAuthMethod extends AbstractAuthMethod {
047    private static final Pattern pemPattern = Pattern.compile("-----BEGIN PRIVATE KEY-----\\n(.*\\n)-----END PRIVATE KEY-----", Pattern.MULTILINE | Pattern.DOTALL);
048    public final int SORT_KEY = 10;
049    private String applicationId;
050    private JWTSigner signer;
051
052    public JWTAuthMethod(final String applicationId, final byte[] privateKey)
053            throws NoSuchAlgorithmException, InvalidKeyException, InvalidKeySpecException {
054        this.applicationId = applicationId;
055
056        byte[] decodedPrivateKey = privateKey;
057        if (privateKey[0] == '-') {
058            decodedPrivateKey = decodePrivateKey(privateKey);
059        }
060        PKCS8EncodedKeySpec spec = new PKCS8EncodedKeySpec(decodedPrivateKey);
061        KeyFactory kf = KeyFactory.getInstance("RSA");
062        PrivateKey key = kf.generatePrivate(spec);
063        this.signer = new JWTSigner(key);
064    }
065
066    public JWTAuthMethod(String applicationId, Path path)
067            throws NoSuchAlgorithmException, InvalidKeyException, InvalidKeySpecException, IOException {
068        this(applicationId, Files.readAllBytes(path));
069    }
070
071    protected static String constructJTI() {
072        return UUID.randomUUID().toString();
073    }
074
075    protected byte[] decodePrivateKey(byte[] data) throws InvalidKeyException {
076        try {
077            String s = new String(data, "UTF-8");
078            Matcher extracter = pemPattern.matcher(s);
079            if (extracter.matches()) {
080                String pemBody = extracter.group(1);
081                return DatatypeConverter.parseBase64Binary(pemBody);
082            } else {
083                throw new InvalidKeyException("Private key should be provided in PEM format!");
084            }
085        } catch (UnsupportedEncodingException exc) {
086            // This should never happen.
087            throw new NexmoUnexpectedException("UTF-8 is an unsupported encoding in this JVM", exc);
088        }
089    }
090
091    @Override
092    public RequestBuilder apply(RequestBuilder request) {
093        String token = this.constructToken(
094                System.currentTimeMillis() / 1000L,
095                constructJTI());
096        request.setHeader("Authorization", "Bearer " + token);
097        return request;
098    }
099
100    protected String constructToken(long iat, String jti) {
101        Map<String, Object> claims = new HashMap<>();
102        claims.put("iat", iat);
103        claims.put("application_id", this.applicationId);
104        claims.put("jti", jti);
105
106        JWTSigner.Options options = new JWTSigner.Options()
107                .setAlgorithm(Algorithm.RS256);
108        String signed = this.signer.sign(claims, options);
109
110        return signed;
111    }
112
113    @Override
114    public int getSortKey() {
115        return this.SORT_KEY;
116    }
117}

024import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
025import com.fasterxml.jackson.annotation.JsonInclude;
026import com.fasterxml.jackson.annotation.JsonProperty;
027import com.fasterxml.jackson.databind.ObjectMapper;
028import com.fasterxml.jackson.databind.SerializationFeature;
029import com.nexmo.client.NexmoUnexpectedException;
030
031import java.io.IOException;
032import java.util.Date;
033
034/**
035 * CallRecord holds the information related to a call. It is obtained using {@link NexmoVoiceClient#listCalls}
036 */
037@JsonInclude(value = JsonInclude.Include.NON_NULL)
038@JsonIgnoreProperties({ "_links" })
039public class CallRecord {
040    private Endpoint to;
041    private Endpoint from;
042
043    private String conversationUuid = null;
044    private CallDirection direction = null;
045    private Integer duration = null;
046    private Date endTime = null;
047    private String network = null;
048    private String price = null;
049    private String rate = null;
050    private Date startTime = null;
051    private CallStatus status = null;
052    private String uuid = null;
053
054    public CallRecord() {}
055
056    public CallRecord(String to, String from) {
057        this(new PhoneEndpoint(to), new PhoneEndpoint(from));
058    }
059
060    public CallRecord(Endpoint to, Endpoint from) {
061        this.to = to;
062        this.from = from;
063    }
064
065    public Endpoint getTo() {
066        return to;
067    }
068
069    public void setTo(Endpoint to) {
070        this.to = to;
071    }
072
073    public Endpoint getFrom() {
074        return from;
075    }
076
077    public void setFrom(Endpoint from) {
078        this.from = from;
079    }
080
081    public String getUuid() {
082        return uuid;
083    }
084
085    public void setUuid(String uuid) {
086        this.uuid = uuid;
087    }
088
089    @JsonProperty("conversation_uuid")
090    public String getConversationUuid() {
091        return conversationUuid;
092    }
093
094    public void setConversationUuid(String conversationUuid) {
095        this.conversationUuid = conversationUuid;
096    }
097
098    public Integer getDuration() {
099        return duration;
100    }
101
102    public void setDuration(Integer duration) {
103        this.duration = duration;
104    }
105
106    @JsonProperty("end_time")
107    public Date getEndTime() {
108        return endTime;
109    }
110
111    public void setEndTime(Date endTime) {
112        this.endTime = endTime;
113    }
114
115    public String getPrice() {
116        return price;
117    }
118
119    public void setPrice(String price) {
120        this.price = price;
121    }
122
123    public String getRate() {
124        return rate;
125    }
126
127    public void setRate(String rate) {
128        this.rate = rate;
129    }
130
131    @JsonProperty("start_time")
132    public Date getStartTime() {
133        return this.startTime;
134    }
135
136    public void setStartTime(Date startTime) {
137        this.startTime = startTime;
138    }
139
140    public CallStatus getStatus() {
141        return status;
142    }
143
144    public void setStatus(CallStatus status) {
145        this.status = status;
146    }
147
148    public CallDirection getDirection() {
149        return direction;
150    }
151
152    public void setDirection(CallDirection direction) {
153        this.direction = direction;
154    }
155
156    public String getNetwork() {
157        return network;
158    }
159
160    public void setNetwork(String network) {
161        this.network = network;
162    }
163
164    public String toString() {
165        return new StringBuilder()
166                .append("<CallRecord ")
167                .append("ID: ").append(this.getUuid()).append(", ")
168                .append("From: ").append(this.getFrom().toLog()).append(", ")
169                .append("To: ").append(this.getTo().toLog()).append(", ")
170                .append("Status: ").append(this.getStatus())
171                .append(">")
172                .toString();
173    }
174
175    public static CallRecord fromJson(String json) {
176        try {
177            ObjectMapper mapper = new ObjectMapper();
178            mapper.disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS);
179            return mapper.readValue(json, CallRecord.class);
180        } catch (IOException jpe) {
181            throw new NexmoUnexpectedException("Failed to produce json from CallRecord object.", jpe);
182        }
183    }
184}

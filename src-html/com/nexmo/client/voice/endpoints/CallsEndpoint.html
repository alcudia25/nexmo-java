022package com.nexmo.client.voice.endpoints;
023
024import com.nexmo.client.HttpWrapper;
025import com.nexmo.client.NexmoClientException;
026import com.nexmo.client.voice.*;
027
028import java.io.IOException;
029
030/**
031 * Represents Nexmo Voice API endpoints below <pre>/calls</pre>, and provides methods to read and update those endpoints.
032 */
033public class CallsEndpoint {
034    private final CreateCallMethod createCall;
035    private final ReadCallMethod readCall;
036    private final ListCallsMethod listCalls;
037    private final ModifyCallMethod modifyCall;
038
039    public CallsEndpoint(HttpWrapper httpWrapper) {
040        this.createCall = new CreateCallMethod(httpWrapper);
041        this.readCall = new ReadCallMethod(httpWrapper);
042        this.listCalls = new ListCallsMethod(httpWrapper);
043        this.modifyCall = new ModifyCallMethod(httpWrapper);
044    }
045
046    /**
047     * Start a call configured by the provided {@link Call} object.
048     * <p>
049     * Requires a {@link com.nexmo.client.auth.JWTAuthMethod} to be provided to the NexmoClient which constructs
050     *
051     * @param callRequest A Call object configuring the call to be created
052     * @return A CallEvent describing the call that was initiated.
053     * @throws IOException          if an error occurs communicating with the Nexmo API
054     * @throws NexmoClientException if an error occurs constructing the Nexmo API request or response
055     */
056    public CallEvent post(Call callRequest) throws IOException, NexmoClientException {
057        return this.createCall.execute(callRequest);
058    }
059
060    /**
061     * List calls which have been conducted by
062     *
063     * @param filter A CallsFilter describing the calls to be searched, or {@code null} for all calls.
064     * @return A CallInfoPage containing a single page of {@link CallInfo} results
065     * @throws IOException          if an error occurs communicating with the Nexmo API
066     * @throws NexmoClientException if an error occurs constructing the Nexmo API request or response
067     */
068    public CallInfoPage get(CallsFilter filter) throws IOException, NexmoClientException {
069        return this.listCalls.execute(filter);
070    }
071
072    /**
073     * List calls which have been conducted by
074     *
075     * @param uuid The uuid of the CallInfo object to be retrieved
076     * @return A CallInfo object describing the state of the call that was made or is in progress
077     * @throws IOException          if an error occurs communicating with the Nexmo API
078     * @throws NexmoClientException if an error occurs constructing the Nexmo API request or response
079     */
080    public CallInfo get(String uuid) throws IOException, NexmoClientException {
081        return this.readCall.execute(uuid);
082    }
083
084
085    /**
086     * Modify an ongoing call.
087     * <p>
088     * Currently this method only supports the "hangup" action.
089     * </p>
090     *
091     * @param uuid The uuid of the CallInfo object to be modified
092     * @return A ModifyCallResponse object describing the state of the call that was modified
093     * @throws IOException          if an error occurs communicating with the Nexmo API
094     * @throws NexmoClientException if an error occurs constructing the Nexmo API request or response
095     */
096    public ModifyCallResponse put(String uuid, String action) throws IOException, NexmoClientException {
097        return this.modifyCall.execute(new CallModifier(uuid, action));
098    }
099}

022package com.nexmo.client.voice.endpoints;
023
024import com.nexmo.client.HttpWrapper;
025import com.nexmo.client.NexmoClientException;
026import com.nexmo.client.NexmoUnexpectedException;
027import com.nexmo.client.auth.AuthMethod;
028import org.apache.commons.logging.Log;
029import org.apache.commons.logging.LogFactory;
030import org.apache.http.HttpResponse;
031import org.apache.http.client.HttpClient;
032import org.apache.http.client.methods.RequestBuilder;
033
034import java.io.IOException;
035import java.io.UnsupportedEncodingException;
036import java.util.HashSet;
037import java.util.Set;
038
039/**
040 * Abstract class to assist in implementing a call against a REST endpoint.
041 * <p>
042 * Concrete implementations must implement {@link #makeRequest(Object)} to construct a {@link RequestBuilder} from the
043 * provided parameterized request object, and {@link #parseResponse(HttpResponse)} to construct the parameterized
044 * {@link HttpResponse} object.
045 * <p>
046 * The REST call is executed by calling {@link #execute(Object)}.
047 *
048 * @param <RequestT> The type of the method-specific request object that will be used to construct an HTTP request
049 * @param <ResultT>  The type of method-specific response object which will be constructed from the returned
050 *                   HTTP response
051 */
052public abstract class AbstractMethod<RequestT, ResultT> implements Method<RequestT, ResultT> {
053    private static final Log LOG = LogFactory.getLog(AbstractMethod.class);
054    private final HttpWrapper httpWrapper;
055    private Set<Class> acceptable;
056
057    public AbstractMethod(HttpWrapper httpWrapper) {
058        this.httpWrapper = httpWrapper;
059    }
060
061    /**
062     * Execute the REST call represented by this method object.
063     *
064     * @param request A RequestT representing input to the REST call to be made
065     * @return A ResultT representing the response from the executed REST call
066     * @throws IOException          if an exception occurs making the REST call
067     * @throws NexmoClientException if there is a problem parsing the HTTP response
068     */
069    // TODO: Consider wrapping IOException in a nexmo-specific transport exception.
070    public ResultT execute(RequestT request) throws IOException, NexmoClientException {
071        try {
072            HttpResponse response = this.httpWrapper.getHttpClient().execute(applyAuth(makeRequest(request)).build());
073            LOG.debug("Response: " + response.getStatusLine());
074            return parseResponse(response);
075        } catch (UnsupportedEncodingException uee) {
076            throw new NexmoUnexpectedException("UTF-8 encoding is not supported by this JVM.", uee);
077        }
078    }
079
080    /**
081     * Apply an appropriate authentication method (specified by {@link #getAcceptableAuthMethods()} to the provided
082     * {@link RequestBuilder}, and return the result.
083     *
084     * @param request A RequestBuilder which has not yet had authentication information applied
085     * @return A RequestBuilder with appropriate authentication information applied (may or not be the same instance as
086     * <pre>request</pre>)
087     * @throws NexmoClientException If no appropriate {@link AuthMethod} is available
088     */
089    protected RequestBuilder applyAuth(RequestBuilder request) throws NexmoClientException {
090        return getAuthMethod(getAcceptableAuthMethods()).apply(request);
091    }
092
093    /**
094     * Utility method for obtaining an appropriate {@link AuthMethod} for this call.
095     *
096     * @param acceptableAuthMethods an array of classes, representing authentication methods that are acceptable for
097     *                              this endpoint
098     * @return An AuthMethod created from one of the provided acceptableAuthMethods.
099     * @throws NexmoClientException If no AuthMethod is available from the provided array of acceptableAuthMethods.
100     */
101    protected AuthMethod getAuthMethod(Class[] acceptableAuthMethods) throws NexmoClientException {
102        if (acceptable == null) {
103            this.acceptable = new HashSet<>();
104            for (Class c : acceptableAuthMethods) {
105                acceptable.add(c);
106            }
107        }
108
109        return this.httpWrapper.getAuthMethods().getAcceptableAuthMethod(acceptable);
110    }
111
112    public void setHttpClient(HttpClient client) {
113        this.httpWrapper.setHttpClient(client);
114    }
115
116    protected abstract Class[] getAcceptableAuthMethods();
117
118    /**
119     * Construct and return a RequestBuilder instance from the provided request.
120     *
121     * @param request A RequestT representing input to the REST call to be made
122     * @return A ResultT representing the response from the executed REST call
123     * @throws NexmoClientException         if a problem is encountered constructing the request or response.
124     * @throws UnsupportedEncodingException if UTF-8 encoding is not supported by the JVM
125     */
126    public abstract RequestBuilder makeRequest(RequestT request)
127            throws NexmoClientException, UnsupportedEncodingException;
128
129    /**
130     * Construct a ResultT representing the contents of the HTTP response returned from the Nexmo Voice API.
131     *
132     * @param response An HttpResponse returned from the Nexmo Voice API
133     * @return A ResultT type representing the result of the REST call
134     * @throws IOException if a problem occurs parsing the response
135     */
136    public abstract ResultT parseResponse(HttpResponse response) throws IOException;
137}

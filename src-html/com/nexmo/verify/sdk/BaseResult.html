022package com.nexmo.verify.sdk;
023
024
025/**
026 * An abstract base class for verification results.
027 *
028 * @author Daniele Ricci
029 */
030public abstract class BaseResult implements java.io.Serializable {
031
032    private static final long serialVersionUID = -2501134793517817181L;
033
034    /**
035     * Verify was successfully submitted to the Nexmo service
036     */
037    public static final int STATUS_OK = 0;
038
039    /**
040     * Verify was rejected due to exceeding the maximum throughput allowed for this account.<br>
041     * Verify can be re-requested after a short delay
042     */
043    public static final int STATUS_THROTTLED = 1;
044
045    /**
046     * Verify was rejected due to incomplete data in the submission request
047     */
048    public static final int STATUS_MISSING_PARAMS = 2;
049
050    /**
051     * Verify was rejected due to an illegal value in one or more elements of the submission request
052     */
053    public static final int STATUS_INVALID_PARAMS = 3;
054
055    /**
056     * Verify was rejected due to receiving invalid account api key and/or secret
057     */
058    public static final int STATUS_INVALID_CREDENTIALS = 4;
059
060    /**
061     * Verify was rejected due to a failure within the Nexmo systems.<br>
062     * Verify can be re-submitted after a short delay
063     */
064    public static final int STATUS_INTERNAL_ERROR = 5;
065
066    /**
067     * Verify was rejected because the Nexmo service was unable to handle this request. eg, the destination was un-routable.
068     */
069    public static final int STATUS_INVALID_REQUEST = 6;
070
071    /**
072     * Verify was rejected because the phone number you tried to submit to has been blacklisted.
073     */
074    public static final int STATUS_NUMBER_BARRED = 7;
075
076    /**
077     * Verify was rejected because your account has been barred, or has not yet been activated
078     */
079    public static final int STATUS_PARTNER_ACCOUNT_BARRED = 8;
080
081    /**
082     * Verify was rejected because your pre-paid balance does not contain enough credit to handle this request.<br>
083     * Please top up your balance before re-submitting this request or subsequent requests.
084     */
085    public static final int STATUS_PARTNER_QUOTA_EXCEEDED = 9;
086
087    /**
088     * Verify was rejected because another verification to the same number was already requested.
089     */
090    public static final int STATUS_ALREADY_REQUESTED = 10;
091
092    /**
093     * The destination number is not in a supported network
094     */
095    public static final int STATUS_UNSUPPORTED_NETWORK = 15;
096
097    /**
098     * The code inserted does not match the expected value
099     */
100    public static final int STATUS_INVALID_CODE = 16;
101
102    /**
103     * A wrong code was provided too many times
104     */
105    public static final int STATUS_WRONG_CODE_THROTTLED = 17;
106
107    /**
108     * There are more than the maximum allowed number of destinations in this request
109     */
110    public static final int STATUS_TOO_MANY_DESTINATIONS = 18;
111
112    /**
113     * There are no matching verification requests
114     */
115    public static final int STATUS_NO_RESPONSE = 101;
116
117    /**
118     * A network error occurred
119     */
120    public static final int STATUS_COMMS_FAILURE = -1;
121
122    private final int status;
123    private final String errorText;
124    private final boolean temporaryError;
125
126    protected BaseResult(final int status,
127                         final String errorText,
128                         final boolean temporaryError) {
129        this.status = status;
130        this.errorText = errorText;
131        this.temporaryError = temporaryError;
132    }
133
134    public int getStatus() {
135        return this.status;
136    }
137
138    public String getErrorText() {
139        return this.errorText;
140    }
141
142    public boolean isTemporaryError() {
143        return this.temporaryError;
144    }
145}

036import org.apache.http.client.methods.HttpPost;
037import org.apache.http.client.utils.URLEncodedUtils;
038import org.apache.http.impl.client.BasicResponseHandler;
039import org.apache.http.message.BasicNameValuePair;

041import org.w3c.dom.Element;

044
045import java.io.IOException;
046import java.io.UnsupportedEncodingException;
047import java.util.List;
048
049public class CheckClient extends LegacyClient {
050    private static final Log log = LogFactory.getLog(CheckClient.class);
051
052    /**
053     * The endpoint path for submitting verification check requests
054     */
055    private static final String ENDPOINT_PATH = "/verify/check/xml";
056
057    /**
058     * Create a new CheckClient.
059     * <p>
060     * This client is used for calling the verify API's check endpoint.
061     *
062     * @param baseUrl The base URL to be used instead of <code>DEFAULT_BASE_URL</code>
063     * @param apiKey Your Nexmo account api key
064     * @param apiSecret Your Nexmo account api secret
065     * @param connectionTimeout over-ride the default connection timeout with this value (in milliseconds)
066     * @param soTimeout over-ride the default read-timeout with this value (in milliseconds)
067     */
068    public CheckClient(final String baseUrl,
069                       final String apiKey,
070                       final String apiSecret,
071                       final int connectionTimeout,
072                       final int soTimeout) {
073
074        super(baseUrl, apiKey, apiSecret, connectionTimeout, soTimeout);
075    }
076
077    public CheckResult check(final String requestId,
078                             final String code) throws IOException, NexmoResponseParseException {
079        return check(requestId, code, null);
080    }
081
082    public CheckResult check(final String requestId,
083                             final String code,
084                             final String ipAddress) throws IOException, NexmoResponseParseException {
085        if (requestId == null || code == null)
086            throw new IllegalArgumentException("request ID and code parameters are mandatory.");
087
088        log.debug("HTTP-Number-Verify-Check Client .. for [ " + requestId + " ] code [ " + code + " ] ");
089
090        List<NameValuePair> params = constructCheckParams(requestId, code);
091
092        if (ipAddress != null)
093            params.add(new BasicNameValuePair("ip_address", ipAddress));
094
095        String verifyCheckBaseUrl = this.makeUrl(ENDPOINT_PATH);
096
097        // Now that we have generated a query string, we can instantiate a HttpClient,
098        // construct a POST method and execute to submit the request
099        String response;
100        HttpPost httpPost = new HttpPost(verifyCheckBaseUrl);
101        try {
102            httpPost.setEntity(new UrlEncodedFormEntity(params, "UTF-8"));
103        } catch (UnsupportedEncodingException uee) {
104            // This should never happen:
105            throw new NexmoUnexpectedException("UTF-8 is an unsupported encoding in this JVM", uee);
106        }
107        String url = verifyCheckBaseUrl + "?" + URLEncodedUtils.format(params, "utf-8");
108
109        try {
110            HttpResponse httpResponse = this.getHttpClient().execute(httpPost);
111            response = new BasicResponseHandler().handleResponse(httpResponse);
112            log.info(".. SUBMITTED NEXMO-HTTP URL [ " + url + " ] -- response [ " + response + " ] ");
113        } catch (HttpResponseException e) {
114            httpPost.abort();
115            log.error("communication failure: ", e);
116
117            // return a COMMS failure ...
118            return new CheckResult(BaseResult.STATUS_COMMS_FAILURE,
119                    null,
120                    0,
121                    null,
122                    "Failed to communicate with NEXMO-HTTP url [ " + url + " ] ..." + e,
123                    true);
124        }
125
126        return parseCheckResponse(response);
127    }
128
129    private List<NameValuePair> constructCheckParams(String requestId, String code) {
130        List<NameValuePair> params = this.constructParams();
131
132        params.add(new BasicNameValuePair("request_id", requestId));
133        params.add(new BasicNameValuePair("code", code));
134        return params;
135    }
136
137    private CheckResult parseCheckResponse(String response) throws NexmoResponseParseException, IOException {
138        Document doc = parseXml(response);
139
140        Element root = doc.getDocumentElement();
141        if (!"verify_response".equals(root.getNodeName()))
142            throw new IOException("No valid response found [ " + response + "] ");
143
144        String eventId = null;
145        int status = -1;
146        float price = -1;
147        String currency = null;
148        String errorText = null;
149
150        NodeList fields = root.getChildNodes();
151        for (int i = 0; i < fields.getLength(); i++) {
152            Node node = fields.item(i);
153            if (node.getNodeType() != Node.ELEMENT_NODE)
154                continue;
155
156            String name = node.getNodeName();
157            if ("event_id".equals(name)) {
158                eventId = XmlUtil.stringValue(node);
159            } else if ("status".equals(name)) {
160                String str = XmlUtil.stringValue(node);
161                try {
162                    if (str != null)
163                        status = Integer.parseInt(str);
164                } catch (NumberFormatException e) {
165                    log.error("xml parser .. invalid value in <status> node [ " + str + " ] ");
166                    status = BaseResult.STATUS_INTERNAL_ERROR;
167                }
168            } else if ("price".equals(name)) {
169                String str = XmlUtil.stringValue(node);
170                try {
171                    if (str != null)
172                        price = Float.parseFloat(str);
173                } catch (NumberFormatException e) {
174                    log.error("xml parser .. invalid value in <price> node [ " + str + " ] ");
175                }
176            } else if ("currency".equals(name)) {
177                currency = XmlUtil.stringValue(node);
178            } else if ("error_text".equals(name)) {
179                errorText = XmlUtil.stringValue(node);
180            }
181        }
182
183        if (status == -1)
184            throw new IOException("Xml Parser - did not find a <status> node");
185
186        // Is this a temporary error ?
187        boolean temporaryError = (status == BaseResult.STATUS_THROTTLED || status == BaseResult.STATUS_INTERNAL_ERROR);
188
189        return new CheckResult(status, eventId, price, currency, errorText, temporaryError);
190    }
191}

046public class VerifyClient extends LegacyClient {
047    private static final Log log = LogFactory.getLog(VerifyClient.class);
048
049    /**
050     * The endpoint path for submitting verification requests
051     */
052    public static final String PATH_VERIFY = "/verify/xml";
053
054    /**
055     * Create a new VerifyClient.
056     * <p>
057     * This client is used for calling the verify API's verify endpoint.
058     *
059     * @param baseUrl The base URL to be used instead of <code>DEFAULT_BASE_URL</code>
060     * @param apiKey Your Nexmo account api key
061     * @param apiSecret Your Nexmo account api secret
062     * @param connectionTimeout over-ride the default connection timeout with this value (in milliseconds)
063     * @param soTimeout over-ride the default read-timeout with this value (in milliseconds)
064     */
065    public VerifyClient(final String baseUrl,
066                        final String apiKey,
067                        final String apiSecret,
068                        final int connectionTimeout,
069                        final int soTimeout) {
070
071        super(baseUrl, apiKey, apiSecret, connectionTimeout, soTimeout);
072    }

097    public VerifyResult verify(final String number,

110
111    public VerifyResult verify(final String number,
112                               final String brand,
113                               final String from,
114                               final int length,
115                               final Locale locale,
116                               final NexmoVerifyClient.LineType type) throws IOException,
117                                                                             NexmoResponseParseException {
118        List<NameValuePair> params = constructVerifyParams(number, brand, from, length, locale, type);
119
120        String verifyBaseUrl = this.makeUrl(PATH_VERIFY);
121
122        // Now that we have generated a query string, we can instantiate a HttpClient,
123        // construct a POST method and execute to submit the request
124        String response;
125        HttpPost httpPost = new HttpPost(verifyBaseUrl);
126        httpPost.setEntity(new UrlEncodedFormEntity(params, "UTF-8"));
127        String url = verifyBaseUrl + "?" + URLEncodedUtils.format(params, "utf-8");
128
129        try {
130            HttpResponse httpResponse = this.getHttpClient().execute(httpPost);
131            response = new BasicResponseHandler().handleResponse(httpResponse);
132            log.info(".. SUBMITTED NEXMO-HTTP URL [ " + url + " ] -- response [ " + response + " ] ");
133        } catch (HttpResponseException e) {
134            httpPost.abort();
135            log.error("communication failure: ", e);
136
137            // return a COMMS failure ...
138            return new VerifyResult(BaseResult.STATUS_COMMS_FAILURE,
139                    null,
140                    "Failed to communicate with NEXMO-HTTP url [ " + url + " ] ..." + e,
141                    true);
142        }
143
144        return parseVerifyResponse(response);
145    }
146
147    protected VerifyResult parseVerifyResponse(String response) throws NexmoResponseParseException {
148        Document doc = parseXml(response);
149
150        Element root = doc.getDocumentElement();
151        if (!"verify_response".equals(root.getNodeName()))
152            throw new NexmoResponseParseException("No valid response found [ " + response + "] ");
153
154        return SharedParsers.parseVerifyResponseXmlNode(root);
155    }
156
157
158    protected List<NameValuePair> constructVerifyParams(
159            String number, String brand, String from, int length, Locale locale, NexmoVerifyClient.LineType type) {
160        if (number == null || brand == null)
161            throw new IllegalArgumentException("number and brand parameters are mandatory.");
162        if (length > 0 && length != 4 && length != 6)
163            throw new IllegalArgumentException("code length must be 4 or 6.");
164
165        log.debug("HTTP-Number-Verify Client .. to [ " + number + " ] brand [ " + brand + " ] ");
166
167        List<NameValuePair> params = this.constructParams();
168        params.add(new BasicNameValuePair("number", number));
169        params.add(new BasicNameValuePair("brand", brand));
170
171        if (from != null)
172            params.add(new BasicNameValuePair("sender_id", from));
173
174        if (length > 0)
175            params.add(new BasicNameValuePair("code_length", String.valueOf(length)));
176
177        if (locale != null)
178            params.add(new BasicNameValuePair("lg",
179                    (locale.getLanguage() + "-" + locale.getCountry()).toLowerCase()));
180
181        if (type != null)
182            params.add(new BasicNameValuePair("require_type", type.toString()));
183        return params;
184    }

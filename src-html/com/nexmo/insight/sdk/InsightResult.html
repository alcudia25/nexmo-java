022package com.nexmo.insight.sdk;
023
024
025import java.math.BigDecimal;
026
027/**
028 * Number insight request result.
029 *
030 * @author Daniele Ricci
031 */
032public class InsightResult {
033
034    /**
035     * Insight was successfully submitted to the Nexmo service
036     */
037    public static final int STATUS_OK = 0;
038
039    /**
040     * Insight was rejected due to exceeding the maximum throughput allowed for this account.<br>
041     * Insight can be re-requested after a short delay
042     */
043    public static final int STATUS_THROTTLED = 1;
044
045    /**
046     * Insight was rejected due to an illegal value in one or more elements of the submission request
047     */
048    public static final int STATUS_INVALID_PARAMS = 3;
049
050    /**
051     * Insight was rejected due to receiving invalid account api key and/or secret
052     */
053    public static final int STATUS_INVALID_CREDENTIALS = 4;
054
055    /**
056     * Insight was rejected due to a failure within the Nexmo systems.<br>
057     * Insight can be re-submitted after a short delay
058     */
059    public static final int STATUS_INTERNAL_ERROR = 5;
060
061    /**
062     * A network error occurred
063     */
064    public static final int STATUS_COMMS_FAILURE = -1;
065
066    private final String requestId;
067    private final String number;
068
069    private final int status;
070    private final String errorText;
071    private final boolean temporaryError;
072
073    private final BigDecimal requestPrice;
074    private final BigDecimal remainingBalance;
075
076    protected InsightResult(final int status,
077                            final String requestId,
078                            final String number,
079                            final BigDecimal requestPrice,
080                            final BigDecimal remainingBalance,
081                            final String errorText,
082                            final boolean temporaryError) {
083        this.status = status;
084        this.errorText = errorText;
085        this.temporaryError = temporaryError;
086        this.requestId = requestId;
087        this.number = number;
088        this.requestPrice = requestPrice;
089        this.remainingBalance = remainingBalance;
090    }
091
092    public int getStatus() {
093        return this.status;
094    }
095
096    public String getErrorText() {
097        return this.errorText;
098    }
099
100    public boolean isTemporaryError() {
101        return this.temporaryError;
102    }
103
104    public String getRequestId() {
105        return this.requestId;
106    }
107
108    public String getNumber() {
109        return this.number;
110    }
111
112    public float getRequestPrice() {
113        return this.requestPrice == null ? 0.0f : this.requestPrice.floatValue();
114    }
115
116    public float getRemainingBalance() {
117        return this.remainingBalance == null ? 0.0f : this.remainingBalance.floatValue();
118    }
119
120    @Override
121    public String toString() {
122        return "InsightResult [status=" + getStatus() + ", requestId=" + this.requestId + "]";
123    }
124}

048/**
049 * Client for talking to the Nexmo REST interface.
050 * <p>
051 * Number insight searches for information about a phone number and returns it to a single URL callback
052 * via a GET/POST request.
053 * To request an insight, call {@link #request}. Please refer to Nexmo website for current insight fees.
054 * Error codes are listed in {@link InsightResult} and also on the documentation website.
055 * More information on method parameters can be found at Nexmo website:
056 * <a href="https://docs.nexmo.com/number-insight">https://docs.nexmo.com/number-insight</a>.
057 *
058 * @author Daniele Ricci
059 */
060public class NexmoInsightClient extends LegacyClient {

121    public InsightResult request(final String number,
122                                 final String callbackUrl) throws IOException,
123                                                                  NexmoResponseParseException {
124        return request(number,
125                       callbackUrl,
126                       null,
127                       -1,
128                       "GET",
129                       null,
130                       null);
131    }

133    public InsightResult request(final String number,
134                                 final String callbackUrl,
135                                 final String[] features,
136                                 final long callbackTimeout,
137                                 final String callbackMethod,
138                                 final String clientRef,
139                                 final String ipAddress) throws IOException,
140                                                                NexmoResponseParseException {
141        if (number == null || callbackUrl == null)
142            throw new IllegalArgumentException("number and callbackUrl parameters are mandatory.");
143        if (callbackTimeout >= 0 && (callbackTimeout < 1000 || callbackTimeout > 30000))
144            throw new IllegalArgumentException("callback timeout must be between 1000 and 30000.");
145
146        log.debug("HTTP-Number-Insight Client .. to [ " + number + " ] ");
147
148        List<NameValuePair> params = constructParams(number, callbackUrl, features, callbackTimeout, callbackMethod, clientRef, ipAddress);
149
150        String insightBaseUrl = this.makeUrl(PATH_INSIGHT);
151
152        String response;
153        HttpPost httpPost = new HttpPost(insightBaseUrl);
154        httpPost.setEntity(new UrlEncodedFormEntity(params, "UTF-8"));
155        String url = insightBaseUrl + "?" + URLEncodedUtils.format(params, "utf-8");
156
157        try {
158            HttpResponse httpResponse = this.httpClient.execute(httpPost);
159            response = new BasicResponseHandler().handleResponse(httpResponse);
160            log.info(".. SUBMITTED NEXMO-HTTP URL [ " + url + " ] -- response [ " + response + " ] ");
161        } catch (HttpResponseException e) {
162            httpPost.abort();
163            log.error("Communication failure", e);
164
165            // return a COMMS failure ...
166            return new InsightResult(InsightResult.STATUS_COMMS_FAILURE,
167                                     null,
168                                     null,
169                                     null,
170                                     null,
171                                     "Failed to communicate with NEXMO-HTTP url [ " + url + " ] ..." + e,
172                                     true);
173        }
174
175        return parseInsightResult(response);
176    }
177
178    private List<NameValuePair> constructParams(String number, String callbackUrl, String[] features, long callbackTimeout, String callbackMethod, String clientRef, String ipAddress) {
179        List<NameValuePair> params = constructParams();
180
181        params.add(new BasicNameValuePair("number", number));
182        params.add(new BasicNameValuePair("callback", callbackUrl));
183
184        if (features != null)
185            params.add(new BasicNameValuePair("features", strJoin(features, ",")));
186
187        if (callbackTimeout >= 0)
188            params.add(new BasicNameValuePair("callback_timeout", String.valueOf(callbackTimeout)));
189
190        if (callbackMethod != null)
191            params.add(new BasicNameValuePair("callback_method", callbackMethod));
192
193        if (ipAddress != null)
194            params.add(new BasicNameValuePair("ip", ipAddress));
195
196        if (clientRef != null)
197            params.add(new BasicNameValuePair("client_ref", clientRef));
198        return params;
199    }

201    private static String strJoin(String[] aArr, String sSep) {
202        StringBuilder sbStr = new StringBuilder();
203        for (int i = 0; i < aArr.length; i++) {
204            if (i > 0)
205                sbStr.append(sSep);
206            sbStr.append(aArr[i]);
207        }
208        return sbStr.toString();
209    }

211    protected InsightResult parseInsightResult(String response) throws NexmoResponseParseException {
212        Document doc = parseXml(response);
213
214        Element root = doc.getDocumentElement();
215        if (!"lookup".equals(root.getNodeName()))
216            throw new NexmoResponseParseException("No valid response found [ " + response + "] ");
217
218        String requestId = null;
219        String number = null;
220        BigDecimal price = null;
221        BigDecimal balance = null;
222        int status = -1;
223        String errorText = null;
224
225        NodeList fields = root.getChildNodes();
226        for (int i = 0; i < fields.getLength(); i++) {
227            Node node = fields.item(i);
228            if (node.getNodeType() != Node.ELEMENT_NODE)
229                continue;
230
231            String name = node.getNodeName();
232            if ("requestId".equals(name)) {
233                requestId = XmlUtil.stringValue(node);
234            } else if ("number".equals(name)) {
235                number = XmlUtil.stringValue(node);
236            } else if ("status".equals(name)) {
237                String str = XmlUtil.stringValue(node);
238                if (str != null)
239                    try {
240                            status = Integer.parseInt(str);
241                    } catch (NumberFormatException e) {
242                        log.error("xml parser .. invalid value in <status> node [ " + str + " ] ");
243                        status = InsightResult.STATUS_INTERNAL_ERROR;
244                    }
245            } else if ("requestPrice".equals(name)) {
246                String str = XmlUtil.stringValue(node);
247                if (str != null)
248                    try {
249                        price = new BigDecimal(str);
250                    } catch (NumberFormatException nfe) {
251                        log.error(String.format("<requestPrice> contained invalid value: %s", str));
252                    }
253            } else if ("remainingBalance".equals(name)) {
254                String str = XmlUtil.stringValue(node);
255                if (str != null)
256                    try {
257                        balance = new BigDecimal(str);
258                    } catch (NumberFormatException nfe) {
259                        log.error(String.format("<balance> contained invalid value: %s", str));
260                    }
261            } else if ("errorText".equals(name)) {
262                errorText = XmlUtil.stringValue(node);
263            }
264        }
265
266        if (status == -1)
267            throw new NexmoResponseParseException("Xml Parser - did not find a <status> node");
268
269        // Is this a temporary error ?
270        boolean temporaryError = (status == InsightResult.STATUS_THROTTLED || status == InsightResult.STATUS_INTERNAL_ERROR);
271
272        return new InsightResult(status,
273                                 requestId,
274                                 number,
275                                 price,
276                                 balance,
277                                 errorText,
278                                 temporaryError);
279    }

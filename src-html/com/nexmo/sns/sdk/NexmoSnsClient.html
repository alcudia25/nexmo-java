128    public SnsServiceResult submit(final Request request) throws Exception {
129        log.info("NEXMO-REST-SNS-SERVICE-CLIENT ... submit request [ " + request.toString() + " ] ");
130        List<NameValuePair> params = constructSubmitParams(request);
131
132        // Now that we have generated a query string, we can instantiate a HttpClient,
133        // construct a POST or GET method and execute to submit the request
134        String response;
135        HttpPost method = new HttpPost(makeUrl(null));
136        method.setEntity(new UrlEncodedFormEntity(params, "UTF-8"));
137        try {
138            HttpResponse httpResponse = this.getHttpClient().execute(method);
139            response = new BasicResponseHandler().handleResponse(httpResponse);
140            log.info(".. SUBMITTED NEXMO-HTTP URL [ " + method.toString() + " ] -- response [ " + response + " ] ");
141        } catch (Exception e) {
142            log.info("communication failure: " + e);
143            method.abort();
144            return new SnsServiceResult(request.getCommand(),
145                                        SnsServiceResult.STATUS_COMMS_FAILURE,
146                                        "Failed to communicate with NEXMO-HTTP url [ " + method.toString() + " ] ..." + e,
147                                        null,
148                                        null);
149        }
150
151        return parseSubmitResponse(response);
152    }
153
154    protected SnsServiceResult parseSubmitResponse(String response) throws Exception {
155        // parse the response doc ...
156
157        /*
158            We receive a response from the api that looks like this, parse the document
159            and turn it into an instance of SnsServiceResult
160
161                <nexmo-sns>
162                    <command>subscribe|publish</command>
163                    <resultCode>0</resultCode>
164                    <resultMessage>OK!</resultMessage>
165                    <transactionId>${transaction-id}</transactionId>
166                    <subscriberArn>${subscriber}</subscriberArn>
167                </nexmo-sns>
168
169        */
170
171        Document doc = parseXml(response);
172
173        String command = null;
174        int resultCode = -1;
175        String resultMessage = null;
176        String transactionId = null;
177        String subscriberArn = null;
178
179        NodeList replies = doc.getElementsByTagName("nexmo-sns");
180        Node reply = replies.item(0);   // If there's more than one reply, we ignore the extras.
181        NodeList nodes = reply.getChildNodes();
182
183        for (int i2=0;i2<nodes.getLength();i2++) {
184            Node node = nodes.item(i2);
185            if (node.getNodeType() == Node.ELEMENT_NODE) {
186                if (node.getNodeName().equals("command")) {
187                    command = XmlUtil.stringValue(node);
188                } else if (node.getNodeName().equals("resultCode")) {
189                    try {
190                        resultCode = XmlUtil.intValue(node);
191                    } catch (Exception e) {
192                        log.error("xml parser .. invalid value in <resultCode> node [ " + XmlUtil.stringValue(node) +
193                                " ] ");
194                        resultCode = SnsServiceResult.STATUS_INTERNAL_ERROR;
195                    }
196                } else if (node.getNodeName().equals("resultMessage")) {
197                    resultMessage = XmlUtil.stringValue(node);
198                } else if (node.getNodeName().equals("transactionId")) {
199                    transactionId = XmlUtil.stringValue(node);
200                } else if (node.getNodeName().equals("subscriberArn")) {
201                    subscriberArn = XmlUtil.stringValue(node);
202                } else
203                    log.error("xml parser .. unknown node found in nexmo-sns [ " + node.getNodeName() + " ] ");
204            }
205        }
206
207        if (resultCode == -1)
208            throw new NexmoResponseParseException("Xml Parser - did not find a <resultCode> node");
209
210        return new SnsServiceResult(command,
211                                    resultCode,
212                                    resultMessage,
213                                    transactionId,
214                                    subscriberArn);
215    }
216
217    private List<NameValuePair> constructSubmitParams(Request request) {
218        // Construct a query string as a list of NameValuePairs
219        List<NameValuePair> params = constructParams();
220        params.add(new BasicNameValuePair("cmd", request.getCommand()));
221        if (request.getQueryParameters() != null)
222            for (Map.Entry<String, String> entry: request.getQueryParameters().entrySet())
223                params.add(new BasicNameValuePair(entry.getKey(), entry.getValue()));
224        return params;
225    }

022package com.nexmo.common.http;
023
024
025import java.util.HashMap;
026import java.util.Map;
027
028import org.apache.http.client.HttpClient;
029import org.apache.http.impl.client.DefaultHttpClient;
030import org.apache.http.impl.conn.tsccm.ThreadSafeClientConnManager;
031import org.apache.http.params.BasicHttpParams;
032import org.apache.http.params.HttpConnectionParams;
033import org.apache.http.params.HttpParams;
034import org.apache.http.params.HttpProtocolParams;
035
036/**
037 * A Helper factory for instantiating HttpClient instances.
038 *
039 * @author Paul Cook
040 */
041public class HttpClientUtils {
042
043    private final static Map<String, HttpClientUtils> instances = new HashMap<>();
044
045    private final ThreadSafeClientConnManager threadSafeClientConnManager;
046
047    private final int connectionTimeout;
048    private final int soTimeout;
049
050    private HttpClientUtils(int connectionTimeout, int soTimeout) {
051        this.connectionTimeout = connectionTimeout;
052        this.soTimeout = soTimeout;
053
054        this.threadSafeClientConnManager = new ThreadSafeClientConnManager();
055        this.threadSafeClientConnManager.setDefaultMaxPerRoute(200);
056        this.threadSafeClientConnManager.setMaxTotal(200);
057    }
058
059    /**
060     * Return an existing or instantiate a new HttpClient factory instance with explicitly specified connection and read timeout values
061     *
062     * @param connectionTimeout the timeout value in milliseconds to use when establishing a new http socket
063     * @param soTimeout the timeout value in milliseconds to wait for a http response before closing the socket
064     *
065     * @return HttpClientUtils an instance of the HttpClient factory primed with the requested timeout values
066     */
067    public static HttpClientUtils getInstance(int connectionTimeout, int soTimeout) {
068        String key = "c-" + connectionTimeout + "-so-" + soTimeout;
069        HttpClientUtils instance = instances.get(key);
070        if (instance == null) {
071            instance = new HttpClientUtils(connectionTimeout, soTimeout);
072            instances.put(key, instance);
073        }
074        return instance;
075    }
076
077    /**
078     * Instantiate a new HttpClient instance that uses the timeout values associated with this factory instance
079     *
080     * @return HttpClient a new HttpClient instance
081     */
082    public HttpClient getNewHttpClient() {
083        HttpParams httpClientParams = new BasicHttpParams();
084        HttpProtocolParams.setUserAgent(httpClientParams, "nexmo-java/2.0.0-prerelease");
085        HttpProtocolParams.setContentCharset(httpClientParams, "UTF-8");
086        HttpProtocolParams.setHttpElementCharset(httpClientParams, "UTF-8");
087        HttpConnectionParams.setConnectionTimeout(httpClientParams, this.connectionTimeout);
088        HttpConnectionParams.setSoTimeout(httpClientParams, this.soTimeout);
089        HttpConnectionParams.setStaleCheckingEnabled(httpClientParams, true);
090        HttpConnectionParams.setTcpNoDelay(httpClientParams, true);
091
092        return new DefaultHttpClient(this.threadSafeClientConnManager, httpClientParams);
093    }
094
095}

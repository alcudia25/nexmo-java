022package com.nexmo.common;
023
024
025import com.nexmo.client.NexmoResponseParseException;
026import com.nexmo.common.http.HttpClientUtils;
027import com.nexmo.common.util.XmlUtil;
028import com.nexmo.security.RequestSigning;
029import org.apache.http.NameValuePair;
030import org.apache.http.client.HttpClient;
031import org.apache.http.message.BasicNameValuePair;
032import org.w3c.dom.Document;
033
034import javax.xml.parsers.DocumentBuilder;
035import javax.xml.parsers.DocumentBuilderFactory;
036import javax.xml.parsers.ParserConfigurationException;
037import java.util.ArrayList;
038import java.util.List;
039import java.util.concurrent.locks.Lock;
040import java.util.concurrent.locks.ReentrantLock;
041
042
043public abstract class LegacyClient {
044    /**
045     * A lock associated with {@link #documentBuilder}.
046     */
047    private final Lock documentBuilderLock = new ReentrantLock();
048
049    /**
050     * Used for parsing XML data.
051     *
052     * Do not use this without locking on {@link #documentBuilderLock}
053     */
054    private DocumentBuilder documentBuilder;
055
056    private HttpClient httpClient;
057
058    private final String baseUrl;
059
060    private final String apiKey;
061
062    private final String apiSecret;
063
064    private final int connectionTimeout;
065
066    private final int soTimeout;
067
068    private final boolean signRequests;
069
070    private final String signatureSecretKey;
071
072    public LegacyClient(final String baseUrl,
073                        final String apiKey,
074                        final String apiSecret,
075                        final int connectionTimeout,
076                        final int soTimeout) {
077        this(baseUrl, apiKey, apiSecret, connectionTimeout, soTimeout, false, null);
078    }
079
080    public LegacyClient(final String baseUrl,
081                        final String apiKey,
082                        final String apiSecret,
083                        final int connectionTimeout,
084                        final int soTimeout,
085                        boolean signRequests,
086                        String signatureKey) {
087
088        // Derive a http and a https version of the supplied base url
089        if (baseUrl == null)
090            throw new IllegalArgumentException("base url is null");
091        String url = baseUrl.trim();
092        String lc = url.toLowerCase();
093        if (!lc.startsWith("https://"))
094            throw new IllegalArgumentException("base url does not start with https://");
095
096        this.baseUrl = url;
097        this.apiKey = apiKey;
098        this.apiSecret = apiSecret;
099        this.connectionTimeout = connectionTimeout;
100        this.soTimeout = soTimeout;
101        this.signRequests = signRequests;
102        this.signatureSecretKey = signatureKey;
103    }
104
105    /**
106     * Parse a provided XML String and return the generated DOM Document.
107     *
108     * @param xml A String containing XML.
109     * @return A Document generated from the parsed XML.
110     * @throws NexmoResponseParseException If there is a problem initializing the XML parser or parsing the XML.
111     */
112    protected Document parseXml(String xml) throws NexmoResponseParseException {
113        // TODO: Maybe an Error subclass for XML initialization errors, as these are serious and unexpected.
114        Document doc;
115        this.documentBuilderLock.lock();
116        try {
117            if (this.documentBuilder == null) {
118                DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
119                this.documentBuilder = documentBuilderFactory.newDocumentBuilder();
120            }
121            doc = XmlUtil.parseXmlString(this.documentBuilder, xml);
122        } catch (ParserConfigurationException e) {
123            throw new NexmoResponseParseException("Exception initialing XML parser", e);
124        } finally {
125            this.documentBuilderLock.unlock();
126        }
127        return doc;
128    }
129
130    protected HttpClient getHttpClient() {
131        if (this.httpClient == null) {
132            this.httpClient = HttpClientUtils.getInstance(this.connectionTimeout, this.soTimeout).getNewHttpClient();
133        }
134        return this.httpClient;
135    }
136
137    protected List<NameValuePair> constructParams() {
138        List<NameValuePair> params = new ArrayList<>();
139
140        params.add(new BasicNameValuePair("api_key", this.apiKey));
141        if (!this.signRequests) {
142            params.add(new BasicNameValuePair("api_secret", this.apiSecret));
143        }
144
145        return params;
146    }
147
148    protected void signParams(List<NameValuePair> params) {
149        if (this.signRequests) {
150            RequestSigning.constructSignatureForRequestParameters(params, this.signatureSecretKey);
151        }
152    }
153
154    protected String makeUrl(final String path) {
155        if (path != null) {
156            return this.baseUrl + path;
157        } else {
158            return this.baseUrl;
159        }
160    }
161
162    // Allowing users of this client to plugin their own HttpClient.
163    public void setHttpClient(HttpClient httpClient) {
164        this.httpClient = httpClient;
165    }
166}

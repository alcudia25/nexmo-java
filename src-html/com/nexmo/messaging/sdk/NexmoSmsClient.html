050/**
051 * Client for talking to the Nexmo API.
052 * <p>
053 * To submit a message, instantiate a NexmoSmsClient, passing the credentials for your Nexmo account on
054 * the constructor. Then instantiate the appropriate {@link com.nexmo.messaging.sdk.messages.Message}
055 * subclass depending on which type of message you are going to submit. The following subclasses are available:
056 *
057 * <ul>
058 * <li>{@link com.nexmo.messaging.sdk.messages.TextMessage}
059 * <li>{@link com.nexmo.messaging.sdk.messages.BinaryMessage}
060 * <li>{@link com.nexmo.messaging.sdk.messages.WapPushMessage}
061 * <li>{@link com.nexmo.messaging.sdk.messages.UnicodeMessage}
062 * </ul>
063 * <p>
064 * Once you have a {@link com.nexmo.messaging.sdk.messages.Message} object, pass it to {@link #submitMessage(Message)}
065 * which will submit the message to the Nexmo API. It returns an array of
066 * {@link com.nexmo.messaging.sdk.SmsSubmissionResult}, with 1 entry for every sms message that was sent.
067 * Text messages greater than 160 characters will require multiple SMS messages to be submitted.
068 * Each entry in this array will contain an individual messageId as well as an individual status detailing the success or reason for failure of each message.
069 * <p>
070 * The list of possible status codes is listed below:
071 * <ul>
072 *  <li>0 = success  -- the message was successfully accepted for delivery by nexmo
073 *  <li>1 = throttled -- You have exceeded the submission capacity allowed on this account, please back-off and re-try
074 *  <li>2 = missing params -- Your request is incomplete and missing some mandatory parameters
075 *  <li>3 = invalid params -- The value of 1 or more parameters is invalid
076 *  <li>4 = invalid credentials -- The api key / secret you supplied is either invalid or disabled
077 *  <li>5 = internal error -- An error has occurred in the nexmo platform whilst processing this message
078 *  <li>6 = invalid message -- The Nexmo platform was unable to process this message, for example, an un-recognized number prefix
079 *  <li>7 = number barred -- The number you are trying to submit to is blacklisted and may not receive messages
080 *  <li>8 = partner account barred -- The api key you supplied is for an account that has been barred from submitting messages
081 *  <li>9 = partner quota exceeded -- Your pre-pay account does not have sufficient credit to process this message
082 *  <li>10 = too many existing binds -- The number of simultaneous connections to the platform exceeds the capabilities of your account
083 *  <li>11 = account not enabled for http -- This account is not provisioned for http / rest submission, you should use SMPP instead
084 *  <li>12 = message too long -- The message length exceeds the maximum allowed
085 *  <li>13 = comms failure -- There was a network failure attempting to contact Nexmo
086 *  <li>14 = Invalid Signature -- The signature supplied with this request was not verified successfully
087 *  <li>15 = invalid sender address -- The sender address was not allowed for this message
088 *  <li>16 = invalid TTL -- The ttl parameter values, or combination of parameters is invalid
089 *  <li>17 = number unreachable -- this destination cannot be delivered to at this time (if reachable=true is specified)
090 *  <li>18 = too many destinations -- There are more than the maximum allowed number of destinations in this request
091 *  <li>19 = Facility Not Allowed - Your request makes use of a facility that is not enabled on your account
092 *  <li>20 = Invalid Message Class - The message class value supplied was out of range (0 - 3)
093 * </ul>
094 * @author  Paul Cook
095 */

182    /**
183     * Send an SMS message.
184     *
185     * This uses the supplied object to construct a request and post it to the Nexmo API.<br>
186     * This method will respond with an array of SmsSubmissionResult objects. Depending on the nature and length of the submitted message, Nexmo may automatically
187     * split the message into multiple sms messages in order to deliver to the handset. For example, a long text sms of greater than 160 chars will need to be split
188     * into multiple 'concatenated' sms messages. The Nexmo service will handle this automatically for you.<br>
189     * The array of SmsSubmissionResult objects will contain a SmsSubmissionResult object for every actual sms that was required to submit the message.
190     * each message can potentially have a different status result, and each message will have a different message id.
191     * Delivery notifications will be generated for each sms message within this set and will be posted to your application containing the appropriate message id.
192     *
193     * @param message The message request object that describes the type of message and the contents to be submitted.
194     *
195     * @return SmsSubmissionResult[] an array of results, 1 object for each sms message that was required to submit this message in its entirety
196     *
197     * @throws NexmoResponseParseException if the HTTP response could not be parsed.
198     * @throws IOException There has been an error attempting to communicate with the Nexmo service (e.g. Network failure).
199     */
200    public SmsSubmissionResult[] submitMessage(Message message) throws IOException, NexmoResponseParseException {
201        return submitMessage(message,  null);
202    }

204    /**
205     * Submit a message submission request object.
206     * This will use the supplied object to construct a request and post it to the Nexmo REST interface.<br>
207     * This method will respond with an array of SmsSubmissionResult objects. Depending on the nature and length of the submitted message, Nexmo may automatically
208     * split the message into multiple sms messages in order to deliver to the handset. For example, a long text sms of greater than 160 chars will need to be split
209     * into multiple 'concatenated' sms messages. The Nexmo service will handle this automatically for you.<br>
210     * The array of SmsSubmissionResult objects will contain a SmsSubmissionResult object for every actual sms that was required to submit the message.
211     * each message can potentially have a different status result, and each message will have a different message id.
212     * Delivery notifications will be generated for each sms message within this set and will be posted to your application containing the appropriate message id.
213     *
214     * @param message The message request object that describes the type of message and the contents to be submitted.
215     * @param validityPeriod The validity period (Time-To-Live) for this message. Specifies the time before this message will be expired if not yet delivered
216     *
217     * @return SmsSubmissionResult[] an array of results, 1 object for each sms message that was required to submit this message in its entirety
218     *
219     * @throws NexmoResponseParseException if the HTTP response could not be parsed.
220     * @throws IOException There has been a general failure either within the Client class, or whilst attempting to communicate with the Nexmo service (eg, Network failure)
221     */
222    public SmsSubmissionResult[] submitMessage(final Message message,
223                                               final ValidityPeriod validityPeriod) throws IOException, NexmoResponseParseException {
224
225        log.debug("HTTP-Message-Submission Client .. from [ " + message.getFrom() + " ] to [ " + message.getTo() + " ] msg [ " + message.getMessageBody() + " ] ");
226
227        // From the Message object supplied, construct an appropriate request to be submitted to the Nexmo REST Service.
228
229        // Determine what 'product' type we are submitting, and select the appropriate endpoint path
230        List<NameValuePair> params = constructParams(message, validityPeriod);
231
232        String baseUrl = makeUrl(SUBMISSION_PATH_SMS);
233
234        // Now that we have generated a query string, we can instantiate a HttpClient,
235        // and construct a POST request:
236
237        String url;
238
239        HttpPost httpPost = new HttpPost(baseUrl);
240        httpPost.setEntity(new UrlEncodedFormEntity(params, "UTF-8"));
241        url = baseUrl + "?" + URLEncodedUtils.format(params, "utf-8");
242
243        HttpResponse httpResponse = this.getHttpClient().execute(httpPost);
244
245        String response;
246        try {
247            response = new BasicResponseHandler().handleResponse(httpResponse);
248            log.info(".. SUBMITTED NEXMO-HTTP URL [ " + url + " ] -- response [ " + response + " ] ");
249        } catch (HttpResponseException e) {
250            httpPost.abort();
251            log.error("communication failure", e);
252
253            // return a COMMS failure ...
254            return new SmsSubmissionResult[] {
255                    new SmsSubmissionResult(SmsSubmissionResult.STATUS_COMMS_FAILURE,
256                            null,
257                            null,
258                            "Failed to communicate with NEXMO-HTTP url [ " + url + " ] ..." + e,
259                            message.getClientReference(),
260                            null,
261                            null,
262                            true,
263                            null,
264                            null)
265            };
266        }
267
268        return parseResponse(response);
269    }

271    protected List<NameValuePair> constructParams(final Message message,
272                                        final ValidityPeriod validityPeriod) {
273        // Determine the type parameter based on the type of Message object.
274        boolean binary = message.getType() == Message.MESSAGE_TYPE_BINARY;
275        boolean unicode = message.isUnicode();
276        boolean wapPush = message.getType() == Message.MESSAGE_TYPE_WAPPUSH;
277
278        String mode = "text";
279
280        if (binary)
281            mode = "binary";
282
283        if (unicode)
284            mode = "unicode";
285
286        if (wapPush)
287            mode = "wappush";
288
289        List<NameValuePair> params = constructParams();
290        params.add(new BasicNameValuePair("from", message.getFrom()));
291        params.add(new BasicNameValuePair("to", message.getTo()));
292        params.add(new BasicNameValuePair("type", mode));
293        if (wapPush) {
294            params.add(new BasicNameValuePair("url", message.getWapPushUrl()));
295            params.add(new BasicNameValuePair("title", message.getWapPushTitle()));
296            if (message.getWapPushValidity() > 0)
297                params.add(new BasicNameValuePair("validity", "" + message.getWapPushValidity()));
298        } else if (binary) {
299            // Binary Message
300            if (message.getBinaryMessageUdh() != null)
301                params.add(new BasicNameValuePair("udh", HexUtil.bytesToHex(message.getBinaryMessageUdh())));
302            params.add(new BasicNameValuePair("body", HexUtil.bytesToHex(message.getBinaryMessageBody())));
303        } else {
304            // Text Message
305            params.add(new BasicNameValuePair("text", message.getMessageBody()));
306        }
307
308        if (message.getClientReference() != null)
309            params.add(new BasicNameValuePair("client-ref", message.getClientReference()));
310
311        params.add(new BasicNameValuePair("status-report-req", "" + message.getStatusReportRequired()));
312
313        if (message.getMessageClass() != null)
314            params.add(new BasicNameValuePair("message-class", "" + message.getMessageClass().getMessageClass()));
315
316        if (message.getProtocolId() != null)
317            params.add(new BasicNameValuePair("protocol-id", "" + message.getProtocolId()));
318
319        if (validityPeriod != null) {
320            if (validityPeriod.getTimeToLive() != null)
321                params.add(new BasicNameValuePair("ttl", "" + validityPeriod.getTimeToLive()));
322            if (validityPeriod.getValidityPeriodHours() != null)
323                params.add(new BasicNameValuePair("ttl-hours", "" + validityPeriod.getValidityPeriodHours()));
324            if (validityPeriod.getValidityPeriodMinutes() != null)
325                params.add(new BasicNameValuePair("ttl-minutes", "" + validityPeriod.getValidityPeriodMinutes()));
326            if (validityPeriod.getValidityPeriodSeconds() != null)
327                params.add(new BasicNameValuePair("ttl-seconds", "" + validityPeriod.getValidityPeriodSeconds()));
328        }
329
330        this.signParams(params);
331
332        return params;
333    }

335    protected SmsSubmissionResult[] parseResponse(String response) throws NexmoResponseParseException {
336        // parse the response doc ...
337
338        /*
339            We receive a response from the api that looks like this, parse the document
340            and turn it into an array of SmsSubmissionResult, one object per <message> node
341
342                <mt-submission-response>
343                    <messages count='x'>
344                        <message>
345                            <to>xxx</to>
346                            <messageId>xxx</messageId>
347                            <status>xx</status>
348                            <errorText>ff</errorText>
349                            <clientRef>xxx</clientRef>
350                            <remainingBalance>##.##</remainingBalance>
351                            <messagePrice>##.##</messagePrice>
352                            <reachability status='x' description='xxx' />
353                            <network>23410</network>
354                        </message>
355                    </messages>
356                </mt-submission-response>
357        */
358
359        List<SmsSubmissionResult> results = new ArrayList<>();
360
361        Document doc = parseXml(response);
362
363        NodeList replies = doc.getElementsByTagName("mt-submission-response");
364        for (int i=0;i<replies.getLength();i++) {
365            Node reply = replies.item(i);
366            NodeList messageLists = reply.getChildNodes();
367            for (int i2=0;i2<messageLists.getLength();i2++) {
368                Node messagesNode = messageLists.item(i2);
369                if (messagesNode.getNodeType() != Node.ELEMENT_NODE)
370                    continue;
371                if (messagesNode.getNodeName().equals("messages")) {
372                    NodeList messages = messagesNode.getChildNodes();
373                    for (int i3=0;i3<messages.getLength();i3++) {
374                        SmsSubmissionResult message = parseMessageXmlNode(messages.item(i3));
375                        if (message != null) {
376                            results.add(message);
377                        }
378                    }
379                }
380            }
381        }
382
383        return results.toArray(new SmsSubmissionResult[results.size()]);
384    }

386    private SmsSubmissionResult parseMessageXmlNode(Node messageNode)
387        throws NexmoResponseParseException {
388        if (messageNode.getNodeType() != Node.ELEMENT_NODE)
389            return null;
390
391        int status = -1;
392        String messageId = null;
393        String destination = null;
394        String errorText = null;
395        String clientReference = null;
396        BigDecimal remainingBalance = null;
397        BigDecimal messagePrice = null;
398        SmsSubmissionReachabilityStatus smsSubmissionReachabilityStatus = null;
399        String network = null;
400
401        NodeList nodes = messageNode.getChildNodes();
402        for (int i4=0;i4<nodes.getLength();i4++) {
403            Node node = nodes.item(i4);
404            if (node.getNodeType() != Node.ELEMENT_NODE)
405                continue;
406            if (node.getNodeName().equals("messageId")) {
407                messageId = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
408            } else if (node.getNodeName().equals("to")) {
409                destination = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
410            } else if (node.getNodeName().equals("status")) {
411                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
412                try {
413                    status = Integer.parseInt(str);
414                } catch (NumberFormatException e) {
415                    log.error("xml parser .. invalid value in <status> node [ " + str + " ] ");
416                    status = SmsSubmissionResult.STATUS_INTERNAL_ERROR;
417                }
418            } else if (node.getNodeName().equals("errorText")) {
419                errorText = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
420            } else if (node.getNodeName().equals("clientRef")) {
421                clientReference = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
422            } else if (node.getNodeName().equals("remainingBalance")) {
423                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
424                try {
425                    if (str != null)
426                        remainingBalance = new BigDecimal(str);
427                } catch (NumberFormatException e) {
428                    log.error("xml parser .. invalid value in <remainingBalance> node [ " + str + " ] ");
429                }
430            } else if (node.getNodeName().equals("messagePrice")) {
431                String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
432                try {
433                    if (str != null)
434                        messagePrice = new BigDecimal(str);
435                } catch (NumberFormatException e) {
436                    log.error("xml parser .. invalid value in <messagePrice> node [ " + str + " ] ");
437                }
438            } else if (node.getNodeName().equals("network")) {
439                network = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
440            } else {
441                log.error("xml parser .. unknown node found in status-return, expected [ messageId, to, status, errorText, clientRef, messagePrice, remainingBalance, reachability, network ] -- found [ " + node.getNodeName() + " ] ");
442            }
443        }
444
445        if (status == -1)
446            throw new NexmoResponseParseException("Xml Parser - did not find a <status> node");
447
448        // Is this a temporary error ?
449        boolean temporaryError = (status == SmsSubmissionResult.STATUS_THROTTLED ||
450                status == SmsSubmissionResult.STATUS_INTERNAL_ERROR ||
451                status == SmsSubmissionResult.STATUS_TOO_MANY_BINDS);
452
453        return new SmsSubmissionResult(status,
454                destination,
455                messageId,
456                errorText,
457                clientReference,
458                remainingBalance,
459                messagePrice,
460                temporaryError,
461                smsSubmissionReachabilityStatus,
462                network);
463    }

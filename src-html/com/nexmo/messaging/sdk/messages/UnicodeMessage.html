022package com.nexmo.messaging.sdk.messages;
023
024
025import com.nexmo.messaging.sdk.messages.parameters.MessageClass;
026
027/**
028 * Represents the details of a unicode text message that is to be submitted via the Nexmo SMS API.
029 * <p>
030 * Unicode messages are required when sending messages to the handset that contain characters that are not
031 * within the GSM character set, for example when sending messages in Arabic or Chinese.
032 *
033 * @author  Paul Cook
034 */
035public class UnicodeMessage extends Message {
036
037    private static final long serialVersionUID = 8327129506926552344L;
038
039    /**
040     * Instantiate a new unicode text-message request<br>
041     * This message will be submitted as a unicode 16 bit text message
042     *
043     * @param from the 'from' address that will be seen on the handset when this message arrives,
044     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
045     * @param to   the phone number of the handset that you wish to send the message to
046     * @param messageBody The text of the message to be sent to the handset
047     */
048    public UnicodeMessage(final String from,
049                          final String to,
050                          final String messageBody) {
051        super(MESSAGE_TYPE_UNICODE,
052              from,
053              to,
054              messageBody,
055              null,
056              null,
057              null,
058              true,
059              false,
060              null,
061              null,
062              0,
063              null,
064              null);
065    }
066
067    /**
068     * Instantiate a new unicode text-message request, exposing all of the available parameters, and optionally requesting a delivery notification<br>
069     * This message will be submitted as a unicode 16 bit text message
070     *
071     * @param from the 'from' address that will be seen on the handset when this message arrives,
072     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
073     * @param to   the phone number of the handset that you wish to send the message to
074     * @param messageBody The text of the message to be sent to the handset
075     * @param clientReference This is a user definable reference that will be stored in the Nexmo messaging records. It will be available in detailed reporting / analytics
076     *                        In order to help with reconciliation of messages
077     * @param statusReportRequired If set to true, then a delivery notification will be requested for this message delivery attempt.
078     *                             Upon receiving notification of delivery or failure from the network, the Nexmo platform will submit a notification to the url configured in your
079     *                             Nexmo REST account that represents the outcome of this message.
080     * @param messageClass (Optional) The message class that is to be applied to this message.
081     * @param protocolId The value of the GSM Protocol ID field to be submitted with this message. Ordinarily this should be left as the default value of 0
082     */
083    public UnicodeMessage(final String from,
084                          final String to,
085                          final String messageBody,
086                          final String clientReference,
087                          final boolean statusReportRequired,
088                          final MessageClass messageClass,
089                          final Integer protocolId) {
090        super(MESSAGE_TYPE_TEXT,
091              from,
092              to,
093              messageBody,
094              null,
095              null,
096              clientReference,
097              true,
098              statusReportRequired,
099              null,
100              null,
101              0,
102              messageClass,
103              protocolId);
104    }
105
106}

022package com.nexmo.messaging.sdk.messages;
023
024
025import com.nexmo.messaging.sdk.messages.parameters.MessageClass;
026
027/**
028 * Represents the details of a message that is to be submitted to the Nexmo SMS API.
029 *
030 * @author  Paul Cook
031 */
032public abstract class Message implements java.io.Serializable {
033
034    private static final long serialVersionUID = 3847700435531116012L;
035
036    /**
037     * Message is a regular TEXT SMS message
038     */
039    public static final int MESSAGE_TYPE_TEXT = 1;
040    /**
041     * Message is a binary SMS message with a custom UDH and binary payload
042     */
043    public static final int MESSAGE_TYPE_BINARY = 2;
044    /**
045     * Message is a wap-push message to send a browsable / downloadable url to the handset
046     */
047    public static final int MESSAGE_TYPE_WAPPUSH = 3;
048    /**
049     * Message is a unicode message, for sending messages in non-latin script to a supported handset
050     */
051    public static final int MESSAGE_TYPE_UNICODE = 4;
052
053    private final int type;
054
055    private final String from;
056    private final String to;
057    private final String messageBody;
058    private final byte[] binaryMessageBody;
059    private final byte[] binaryMessageUdh;
060    private final String clientReference;
061    private final boolean unicode;
062    private final boolean statusReportRequired;
063    private final String wapPushUrl;
064    private final String wapPushTitle;
065    private final int wapPushValidity;
066    private final MessageClass messageClass;
067    private final Integer protocolId;
068
069    /**
070     * Instantiate a new message request.<br>
071     * This constructor exposes the full range of possible parameters and is not for general use
072     * Instead, it is accessed via super() in the constructors of various sub-classes that expose a relevant
073     * sub-set of the available parameters
074     *
075     * @param type the type of message will influence the makeup of the request we post to the Nexmo server, and also the action taken by the Nexmo server in response to this message
076     * @param from the 'from' address that will be seen on the handset when this message arrives,
077     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
078     * @param to   the phone number of the handset that you wish to send the message to
079     * @param messageBody The text of the message to be sent to the handset
080     * @param binaryMessageBody The raw binary message data to be sent to a handset.
081     *                          This api, and the Nexmo messaging service will send this data 'as-is' (in conjunction with the binary UDH) and will not make any corrections.
082     *                          so you should ensure that it is a correctly constructed message
083     * @param binaryMessageUdh Most binary content will require a UserDataHeader portion of the message containing commands to enable the handset to interpret the binary data
084     *                         (for example, a binary ringtone, a wap-push, OverTheAir configuration, etc).
085     *                         Additionally, if you are sending a long text message as multiple concatenated messages and are performing this operation manually rather than
086     *                         using the automated long messaging handling in the Nexmo messaging service, then you will need to construct and include here an appropriate
087     *                         UserDataHeader field that describes the segmentation/re-assembly fields required to successfully concatenate multiple short messages.
088     * @param clientReference This is a user definable reference that will be stored in the Nexmo messaging records. It will be available in detailed reporting / analytics
089     *                        In order to help with reconciliation of messages
090     * @param unicode This flag is set to true if the message needs to be submitted as a unicode message. This would be for scenario's where the message contains text that does not
091     *                fit within the Latin GSM alphabet. Examples would be messages to be sent in non-western scripts, such as Arabic, Kanji, Chinese, etc.
092     * @param statusReportRequired If set to true, then a delivery notification will be requested for this message delivery attempt.
093     *                             Upon receiving notification of delivery or failure from the network, the Nexmo platform will submit a notification to the url configured in your
094     *                             Nexmo REST account that represents the outcome of this message.
095     * @param wapPushUrl This is the url that will be submitted to the handset and will appear as a browsable message in the Inbox.
096     * @param wapPushTitle This is the title that will be associated with the url being submitted to the handset
097     * @param wapPushValidity This is the length of time (in seconds) that the message will be available for once delivered to the handset.
098     *                        Once this time has expired, the url will no longer be visible on the handset to be browsed (Subject to handset compatibility)
099     * @param messageClass (Optional) The message class that is to be applied to this message.
100     * @param protocolId The value of the GSM Protocol ID field to be submitted with this message. Ordinarily this should be left as the default value of 0
101     */
102    protected Message(final int type,
103                      final String from,
104                      final String to,
105                      final String messageBody,
106                      final byte[] binaryMessageBody,
107                      final byte[] binaryMessageUdh,
108                      final String clientReference,
109                      final boolean unicode,
110                      final boolean statusReportRequired,
111                      final String wapPushUrl,
112                      final String wapPushTitle,
113                      final int wapPushValidity,
114                      final MessageClass messageClass,
115                      final Integer protocolId) {
116        this.type = type;
117
118        this.from = from;
119        this.to = to;
120        this.messageBody = messageBody;
121        this.binaryMessageBody = binaryMessageBody;
122        this.binaryMessageUdh = binaryMessageUdh;
123        this.clientReference = clientReference;
124        this.unicode = unicode;
125        this.statusReportRequired = statusReportRequired;
126        this.wapPushUrl = wapPushUrl;
127        this.wapPushTitle = wapPushTitle;
128        this.wapPushValidity = wapPushValidity;
129        this.messageClass = messageClass;
130        this.protocolId = protocolId;
131    }
132
133    /**
134     * @return int the type of message will influence the makeup of the request we post to the Nexmo server, and also the action taken by the Nexmo server in response to this message
135     */
136    public int getType() {
137        return this.type;
138    }
139
140    /**
141     * @return String the 'from' address that will be seen on the handset when this message arrives,
142     *                typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
143     */
144    public String getFrom() {
145        return this.from;
146    }
147
148    /**
149     * @return String the phone number of the handset that you wish to send the message to
150     */
151    public String getTo() {
152        return this.to;
153    }
154
155    /**
156     * @return String The text of the message to be sent to the handset
157     */
158    public String getMessageBody() {
159        return this.messageBody;
160    }
161
162    /**
163     * @return byte[] The raw binary message data to be sent to a handset.
164     *                This api, and the Nexmo messaging service will send this data 'as-is' (in conjunction with the binary UDH) and will not make any corrections.
165     *                so you should ensure that it is a correctly constructed message
166     */
167    public byte[] getBinaryMessageBody() {
168        return this.binaryMessageBody == null ? null : this.binaryMessageBody.clone();
169    }
170
171    /**
172     * @return byte[] Most binary content will require a UserDataHeader portion of the message containing commands to enable the handset to interpret the binary data
173     *                (for example, a binary ringtone, a wap-push, OverTheAir configuration, etc).
174     *                Additionally, if you are sending a long text message as multiple concatenated messages and are performing this operation manually rather than
175     *                using the automated long messaging handling in the Nexmo messaging service, then you will need to construct and include here an appropriate
176     *                UserDataHeader field that describes the segmentation/re-assembly fields required to successfully concatenate multiple short messages.
177     */
178    public byte[] getBinaryMessageUdh() {
179        return this.binaryMessageUdh == null ? null : this.binaryMessageUdh.clone();
180    }
181
182    /**
183     * @return String This is a user definable reference that will be stored in the Nexmo messaging records. It will be available in detailed reporting / analytics
184     *                In order to help with reconciliation of messages
185     */
186    public String getClientReference() {
187        return this.clientReference;
188    }
189
190    /**
191     * @return boolean This flag is set to true if the message needs to be submitted as a unicode message. This would be for scenario's where the message contains text that does not
192     *                 fit within the Latin GSM alphabet. Examples would be messages to be sent in non-western scripts, such as Arabic, Kanji, Chinese, etc.
193     */
194    public boolean isUnicode() {
195        return this.unicode;
196    }
197
198    /**
199     * @return boolean If set to true, then a delivery notification will be requested for this message delivery attempt.
200     *                 Upon receiving notification of delivery or failure from the network, the Nexmo platform will submit a notification to the url configured in your
201     *                 Nexmo REST account that represents the outcome of this message.
202     */
203    public boolean getStatusReportRequired() {
204        return this.statusReportRequired;
205    }
206
207    /**
208     * @return String This is the url that will be submitted to the handset and will appear as a browsable message in the Inbox.
209     */
210    public String getWapPushUrl() {
211        return this.wapPushUrl;
212    }
213
214    /**
215     * @return String This is the title that will be associated with the url being submitted to the handset
216     */
217    public String getWapPushTitle() {
218        return this.wapPushTitle;
219    }
220
221    /**
222     * @return int This is the length of time (in seconds) that the message will be available for once delivered to the handset.
223     *             Once this time has expired, the url will no longer be visible on the handset to be browsed (Subject to handset compatibility)
224     */
225    public int getWapPushValidity() {
226        return this.wapPushValidity;
227    }
228
229    /**
230     * @return com.nexmo.messaging.sdk.messages.parameters.MessageClass The message class that is to be applied to this message.
231     */
232    public MessageClass getMessageClass() {
233        return this.messageClass;
234    }
235
236    /**
237     * @return Integer The value of the GSM Protocol ID field to be submitted with this message. Ordinarily this should be left as the default value of 0
238     */
239    public Integer getProtocolId() {
240        return this.protocolId;
241    }
242
243}

022package com.nexmo.messaging.sdk.messages;
023
024
025import com.nexmo.messaging.sdk.messages.parameters.MessageClass;
026
027/**
028 * Represents the details of a binary message that is to be submitted via the Nexmo REST API.
029 *
030 * @author  Paul Cook
031 */
032public class BinaryMessage extends Message {
033
034    private static final long serialVersionUID = -5301775803396025570L;
035
036    /**
037     * Instantiate a new binary sms message request.
038     *
039     * @param from the 'from' address that will be seen on the handset when this message arrives,
040     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
041     * @param to   the phone number of the handset that you wish to send the message to
042     * @param binaryMessageBody The raw binary message data to be sent to a handset.
043     *                          This api, and the Nexmo messaging service will send this data 'as-is' (in conjunction with the binary UDH) and will not make any corrections.
044     *                          so you should ensure that it is a correctly constructed message
045     * @param binaryMessageUdh Most binary content will require a UserDataHeader portion of the message containing commands to enable the handset to interpret the binary data
046     *                         (for example, a binary ringtone, a wap-push, OverTheAir configuration, etc).
047     *                         Additionally, if you are sending a long text message as multiple concatenated messages and are performing this operation manually rather than
048     *                         using the automated long messaging handling in the Nexmo messaging service, then you will need to construct and include here an appropriate
049     *                         UserDataHeader field that describes the segmentation/re-assembly fields required to successfully concatenate multiple short messages.
050     */
051    public BinaryMessage(final String from,
052                         final String to,
053                         final byte[] binaryMessageBody,
054                         final byte[] binaryMessageUdh) {
055        super(MESSAGE_TYPE_BINARY,
056              from,
057              to,
058              null,
059              binaryMessageBody,
060              binaryMessageUdh,
061              null,
062              false,
063              false,
064              null,
065              null,
066              0,
067              null,
068              null);
069    }
070
071    /**
072     * Instantiate a new binary sms message request, attaching an optional client reference, message-class or TTL, and optionally requesting a delivery notification.
073     *
074     * @param from the 'from' address that will be seen on the handset when this message arrives,
075     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
076     * @param to   the phone number of the handset that you wish to send the message to
077     * @param binaryMessageBody The raw binary message data to be sent to a handset.
078     *                          This api, and the Nexmo messaging service will send this data 'as-is' (in conjunction with the binary UDH) and will not make any corrections.
079     *                          so you should ensure that it is a correctly constructed message
080     * @param binaryMessageUdh Most binary content will require a UserDataHeader portion of the message containing commands to enable the handset to interpret the binary data
081     *                         (for example, a binary ringtone, a wap-push, OverTheAir configuration, etc).
082     *                         Additionally, if you are sending a long text message as multiple concatenated messages and are performing this operation manually rather than
083     *                         using the automated long messaging handling in the Nexmo messaging service, then you will need to construct and include here an appropriate
084     *                         UserDataHeader field that describes the segmentation/re-assembly fields required to successfully concatenate multiple short messages.
085     * @param clientReference This is a user definable reference that will be stored in the Nexmo messaging records. It will be available in detailed reporting / analytics
086     *                        In order to help with reconciliation of messages
087     * @param statusReportRequired If set to true, then a delivery notification will be requested for this message delivery attempt.
088     *                             Upon receiving notification of delivery or failure from the network, the Nexmo platform will submit a notification to the url configured in your
089     *                             Nexmo REST account that represents the outcome of this message.
090     * @param messageClass (Optional) The message class that is to be applied to this message.
091     * @param protocolId The value of the GSM Protocol ID field to be submitted with this message. Ordinarily this should be left as the default value of 0
092     */
093    public BinaryMessage(final String from,
094                         final String to,
095                         final byte[] binaryMessageBody,
096                         final byte[] binaryMessageUdh,
097                         final String clientReference,
098                         final boolean statusReportRequired,
099                         final MessageClass messageClass,
100                         final Integer protocolId) {
101        super(MESSAGE_TYPE_BINARY,
102              from,
103              to,
104              null,
105              binaryMessageBody,
106              binaryMessageUdh,
107              clientReference,
108              false,
109              statusReportRequired,
110              null,
111              null,
112              0,
113              messageClass,
114              protocolId);
115    }
116
117}

022package com.nexmo.messaging.sdk.messages;
023
024
025/**
026 * Represents the details of a wap-push message that is to be submitted via the Nexmo REST api.
027 *
028 * @author  Paul Cook
029 */
030public class WapPushMessage extends Message {
031
032    private static final long serialVersionUID = -8412926886570923963L;
033
034    /**
035     * Instantiate a new wap-push message request, to submit a browsable / downloadable URL to the handset
036     *
037     * @param from the 'from' address that will be seen on the handset when this message arrives,
038     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
039     * @param to   the phone number of the handset that you wish to send the message to
040     * @param wapPushUrl This is the url that will be submitted to the handset and will appear as a browsable message in the Inbox.
041     * @param wapPushTitle This is the title that will be associated with the url being submitted to the handset
042     */
043    public WapPushMessage(final String from,
044                          final String to,
045                          final String wapPushUrl,
046                          final String wapPushTitle) {
047        super(MESSAGE_TYPE_WAPPUSH,
048              from,
049              to,
050              null,
051              null,
052              null,
053              null,
054              false,
055              false,
056              wapPushUrl,
057              wapPushTitle,
058              0,
059              null,
060              null);
061    }
062
063    /**
064     * Instantiate a new wap-push message request, to submit a browsable / downloadable URL to the handset,
065     * attaching an optional client reference, and optionally requesting a delivery notification
066     *
067     * @param from the 'from' address that will be seen on the handset when this message arrives,
068     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
069     * @param to   the phone number of the handset that you wish to send the message to
070     * @param clientReference This is a user definable reference that will be stored in the Nexmo messaging records. It will be available in detailed reporting / analytics
071     *                        In order to help with reconciliation of messages
072     * @param statusReportRequired If set to true, then a delivery notification will be requested for this message delivery attempt.
073     *                             Upon receiving notification of delivery or failure from the network, the Nexmo platform will submit a notification to the url configured in your
074     *                             Nexmo REST account that represents the outcome of this message.
075     * @param wapPushUrl This is the url that will be submitted to the handset and will appear as a browsable message in the Inbox.
076     * @param wapPushTitle This is the title that will be associated with the url being submitted to the handset
077     * @param wapPushValidity This is the length of time (in seconds) that the message will be available for once delivered to the handset.
078     *                        Once this time has expired, the url will no longer be visible on the handset to be browsed (Subject to handset compatibility)
079     */
080    public WapPushMessage(final String from,
081                          final String to,
082                          final String clientReference,
083                          final boolean statusReportRequired,
084                          final String wapPushUrl,
085                          final String wapPushTitle,
086                          final int wapPushValidity) {
087        super(MESSAGE_TYPE_WAPPUSH,
088              from,
089              to,
090              null,
091              null,
092              null,
093              clientReference,
094              false,
095              statusReportRequired,
096              wapPushUrl,
097              wapPushTitle,
098              wapPushValidity,
099              null,
100              null);
101    }
102
103}

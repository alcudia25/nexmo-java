022package com.nexmo.security;
023
024
025import java.io.UnsupportedEncodingException;
026import java.security.MessageDigest;
027import java.util.List;
028import java.util.Map;
029import java.util.TreeMap;
030
031import javax.servlet.http.HttpServletRequest;
032
033import com.nexmo.client.NexmoUnexpectedException;
034import org.apache.commons.logging.Log;
035import org.apache.commons.logging.LogFactory;
036import org.apache.http.NameValuePair;
037import org.apache.http.message.BasicNameValuePair;
038
039import com.nexmo.common.util.MD5Util;
040
041/**
042 * A helper class for generating or verifying MD5 signatures when signing REST requests for submission to Nexmo.
043 *
044 * @author  Paul Cook
045 */
046public class RequestSigning implements SecurityConstants {
047
048    private static Log log = LogFactory.getLog(RequestSigning.class);
049
050    /**
051     * Signs a set of request parameters.
052     * <p>
053     * Generates additional parameters to represent the timestamp and generated signature.
054     * Uses the supplied pre-shared secret key to generate the signature.
055     *
056     * @param params List of NameValuePair instances containing the query parameters for the request that is to be signed
057     * @param secretKey the pre-shared secret key held by the client
058     *
059     */
060    public static void constructSignatureForRequestParameters(List<NameValuePair> params, String secretKey) {
061        constructSignatureForRequestParameters(params, secretKey, System.currentTimeMillis() / 1000);
062    }
063
064    /**
065     * Signs a set of request parameters.
066     * <p>
067     * Generates additional parameters to represent the timestamp and generated signature.
068     * Uses the supplied pre-shared secret key to generate the signature.
069     *
070     * @param params List of NameValuePair instances containing the query parameters for the request that is to be signed
071     * @param secretKey the pre-shared secret key held by the client
072     * @param currentTimeSeconds the current time in seconds since 1970-01-01
073     *
074     */
075     protected static void constructSignatureForRequestParameters(
076            List<NameValuePair> params, String secretKey, long currentTimeSeconds) {
077        // First, inject a 'timestamp=' parameter containing the current time in seconds since Jan 1st 1970
078        params.add(new BasicNameValuePair(PARAM_TIMESTAMP, Long.toString(currentTimeSeconds)));
079
080        Map<String, String> sortedParams = new TreeMap<>();
081        for (NameValuePair param: params) {
082            String name = param.getName();
083            String value = param.getValue();
084            if (name.equals(PARAM_SIGNATURE))
085                continue;
086            if (value == null)
087                value = "";
088            if (!value.trim().equals(""))
089                sortedParams.put(name, value);
090        }
091
092        // Now, walk through the sorted list of parameters and construct a string
093        StringBuilder sb = new StringBuilder();
094        for (Map.Entry<String, String> param: sortedParams.entrySet()) {
095            String name = param.getKey();
096            String value = param.getValue();
097            sb.append("&").append(clean(name)).append("=").append(clean(value));
098        }
099
100        // Now, append the secret key, and calculate an MD5 signature of the resultant string
101        sb.append(secretKey);
102
103        String str = sb.toString();
104
105        String md5 = "no signature";
106        try {
107            md5 = MD5Util.calculateMd5(str);
108        } catch (Exception e) {
109            log.error("error...", e);
110        }
111
112        log.debug("SECURITY-KEY-GENERATION -- String [ " + str + " ] Signature [ " + md5 + " ] ");
113
114        params.add(new BasicNameValuePair(PARAM_SIGNATURE, md5));
115    }
116
117    /**
118     * Verifies the signature in an HttpServletRequest.
119     *
120     * @param request The HttpServletRequest to be verified
121     * @param secretKey The pre-shared secret key used by the sender of the request to create the signature
122     *
123     * @return true if the signature is correct for this request and secret key.
124     */
125    public static boolean verifyRequestSignature(HttpServletRequest request, String secretKey) {
126        return verifyRequestSignature(request, secretKey, System.currentTimeMillis());
127    }
128
129    /**
130     * Verifies the signature in an HttpServletRequest.
131     *
132     * @param request The HttpServletRequest to be verified
133     * @param secretKey The pre-shared secret key used by the sender of the request to create the signature
134     * @param currentTimeMillis The current time, in milliseconds.
135     *
136     * @return true if the signature is correct for this request and secret key.
137     */
138     protected static boolean verifyRequestSignature(HttpServletRequest request,
139                                                     String secretKey,
140                                                     long currentTimeMillis) {
141        // identify the signature supplied in the request ...
142        String suppliedSignature = request.getParameter(PARAM_SIGNATURE);
143        if (suppliedSignature == null)
144            return false;
145
146        // Firstly, extract the timestamp parameter and verify that it is within 5 minutes of 'current time'
147        String timeString = request.getParameter(PARAM_TIMESTAMP);
148        long time = -1;
149        try {
150            if (timeString != null)
151                time = Long.parseLong(timeString) * 1000;
152        } catch (NumberFormatException e) {
153            log.error("Error parsing 'time' parameter [ " + timeString + " ]", e);
154            time = 0;
155        }
156        long diff = currentTimeMillis - time;
157        if (diff > MAX_ALLOWABLE_TIME_DELTA || diff < -MAX_ALLOWABLE_TIME_DELTA) {
158            log.warn("SECURITY-KEY-VERIFICATION -- BAD-TIMESTAMP ... Timestamp [ " + time + " ] delta [ " + diff + " ] max allowed delta [ " + -MAX_ALLOWABLE_TIME_DELTA + " ] ");
159            return false;
160        }
161
162        // Next, construct a sorted list of the name-value pair parameters supplied in the request, excluding the signature parameter
163        Map<String, String> sortedParams = new TreeMap<>();
164        for (Map.Entry<String, String[]> entry: request.getParameterMap().entrySet()) {
165            String name = entry.getKey();
166            String value = entry.getValue()[0];
167            log.info("" + name + " = " + value);
168            if (name.equals(PARAM_SIGNATURE))
169                continue;
170            if (value == null || value.trim().equals("")) {
171                continue;
172            }
173            sortedParams.put(name, value);
174        }
175
176        // walk this sorted list of parameters and construct a string
177        StringBuilder sb = new StringBuilder();
178        for (Map.Entry<String, String> param: sortedParams.entrySet()) {
179            String name = param.getKey();
180            String value = param.getValue();
181            sb.append("&").append(clean(name)).append("=").append(clean(value));
182        }
183
184        // append the secret key and calculate an md5 signature of the resultant string
185        sb.append(secretKey);
186
187        String str = sb.toString();
188
189        String md5;
190        try {
191            md5 = MD5Util.calculateMd5(str);
192        } catch (Exception e) {
193            log.error("error...", e);
194            return false;
195        }
196
197        log.info("SECURITY-KEY-VERIFICATION -- String [ " + str + " ] Signature [ " + md5 + " ] SUPPLIED SIGNATURE [ " + suppliedSignature + " ] ");
198
199        // verify that the supplied signature matches generated one
200        // use MessageDigest.isEqual as an alternative to String.equals() to defend against timing based attacks
201        try {
202            if (!MessageDigest.isEqual(md5.getBytes("UTF-8"), suppliedSignature.getBytes("UTF-8")))
203                return false;
204        } catch (UnsupportedEncodingException e) {
205            throw new NexmoUnexpectedException("Failed to decode signature as UTF-8", e);
206        }
207
208        return true;
209    }
210
211    public static String clean(String str) {
212        return str == null ? null : str.replaceAll("[=&]", "_");
213    }
